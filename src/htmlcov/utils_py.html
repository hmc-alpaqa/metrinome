17def round_expression(expr: str, digits: int) -> str: 

18 """ 

19 Round all of the numbers in an expression to a given number of decimal places. 

20 

21 Take some sympy expression represented as a string 

22 and round all numbers to the given number of decimal places. 

23 """ 

24 reg_exp = r"([0-9]*)\.([0-9]*)" 

25 

26 def replace_with_rounded(match): 

27 return match.groups()[0] + "." + match.groups()[1][0:digits] 

28 

29 return re.sub(reg_exp, replace_with_rounded, expr)

32def classify(expr: str, var="n") -> str: 

33 """ 

34 Given an arbitrary expression represented as a string, return a value indicating its 'class'. 

35 

36 Note: the string cannot have other variables (e.g. we cannot classify '2m' in terms of 'n'). 

37 

38 Exponentiation is represented by '^'. 

39 

40 Const: [Value] 

41 PolyDeg: [Highest Degree of Polynomial] 

42 ExpBase: [Return the base of expression of the form a^x] 

43 """ 

44 if expr == '': 

45 return "Const:0" 

46 

47 try: 

48 val = float(str(expr)) 

49 return f"Const:{val}" 

50 except ValueError: 

51 # If we have anything to the power 'n', then it is exponential. 

52 val = is_exponential(expr, var) 

53 if val is None: 

54 degree = get_degree(expr, var) 

55 return f"PolyDeg:{degree}" 

56 

57 return f"ExpBase:{val}"

60def get_solution_from_roots(roots): 

61 """Return the solution to a recurrence relation given roots of the characteristic equation.""" 

62 # Round to 4 digits. 

63 roots = (complex(round(root.real, 6), round(root.imag, 6)) for root in roots) 

64 

65 # Compute the multiplicy of each root. 

66 roots_with_multiplicities = Counter(roots) 

67 

68 # Compute the coefficients of a_n as a list. 

69 solution = [] 

70 for root in roots_with_multiplicities.keys(): 

71 for i in range(roots_with_multiplicities[root]): 

72 if root == 1: 

73 solution += [sympify(f"(n**{i})")] 

74 else: 

75 solution += [sympify(f"(n**{i})*{root}**n")] 

76 

77 return solution

80def get_recurrence_solution(recurrence: str): 

81 """ 

82 Return the coefficients to a homogeneous linear recurrence relation. 

83 

84 This is represented as a string of the format 

85 

86 c_0*f(n) + c_1*f(n-1) + ... + c_k*f(n-k) 

87 

88 by finding the roots of the characteristic equation. 

89 """ 

90 # Define symbolic terms 

91 # n = symbols('n') 

92 # f = Function('f') 

93 

94 # Regular expression to parse the recurrence relation expression. 

95 # RE matches a particular term: Coefficient + f(something) 

96 match_function = r'f\([ a-zA-Z0-9-+]*\)' 

97 match_obj = re.findall(rf'([ +-.0-9]*)(\*)({match_function})', recurrence) 

98 coeffs: List[float] = [] 

99 for match in match_obj: 

100 coeffs += [float(match[0].replace(" ", ""))] 

101 

102 # Normalize such that leading coefficient is 1. 

103 leading_coeff = coeffs[0] 

104 coeffs = list(map(lambda coeff: coeff / leading_coeff, coeffs)) 

105 

106 # Find the roots of the characteristic equation 

107 # through arbitrary precision root finding function. 

108 roots = polyroots(coeffs, maxsteps=200, extraprec=200) 

109 

110 return get_solution_from_roots(roots)

113def get_taylor_coeffs(func, num_coeffs: int): 

114 """Given an arbitrary rational function, get its Taylor series coefficients.""" 

115 t_var = symbols('t') 

116 series_list = str(series(func, x=t_var, x0=0, n=num_coeffs)).split('+')[::-1] 

117 first_element = series_list[0] 

118 first_power = re.search(r"\*\*([0-9]*)", str(first_element)) 

119 if first_power is not None: 

120 taylor_coeffs = [sympify(f).subs(t_var, 1) for f in series_list] 

121 return taylor_coeffs 

122 

123 return None

126def is_exponential(term: str, var='n'): 

127 """If an expression contains an exponential, return its base. Otherwise, return None.""" 

128 # either ^n or ^(num*n) 

129 num = "([0-9][0-9]*[.][0-9]*)|([.][0-9][0-9]*)|([0-9][0-9]*)" 

130 search_string = rf"({num})\^{var}" 

131 results = re.findall(search_string, term) 

132 max_base = None 

133 if results: 

134 max_base = max(map(lambda res: float(res[0]), results)) 

135 

136 search_string_with_parens = rf"({num})\^\(({num})?\*{var}\)" 

137 results = re.findall(search_string_with_parens, term) 

138 if results: 

139 # a^(bn) should be classified as a^b 

140 

141 def func(res): 

142 return float(res[0])**float(res[4]) 

143 

144 new_base = max(map(func, results)) 

145 if new_base is not None: 

146 if max_base is None or new_base > max_base: 

147 max_base = new_base 

148 

149 return max_base

152def get_degree(term: str, var="n") -> float: 

153 """If an expression is a polynomial, return the degree. Otherwise, return 0.""" 

154 num = "([0-9][0-9]*[.][0-9]*)|([.][0-9][0-9]*)|([0-9][0-9]*)" 

155 regexp = rf"{var}\^({num})" 

156 res = re.findall(regexp, term) 

157 found_deg = 0. 

158 if res: 

159 found_deg = max(map(lambda x: float(x[0]), res)) 

160 

161 if found_deg == 0. and re.search(var, term) is not None: 

162 return 1. 

163 

164 return found_deg

167def big_o(terms): 

168 """ 

169 Compute the big O of some expression in terms of 'n'. 

170 

171 The terms should be a list of expressions represented as strings. 

172 """ 

173 n_var = symbols('n') 

174 if len(terms) == 0: 

175 return 0.0 

176 

177 if len(terms) == 1: 

178 

179 return terms[0] 

180 

181 term_one = terms[0] 

182 term_two = terms[1] 

183 lim = limit(Abs(sympify(term_two) / sympify(term_one)), n_var, float('inf')) 

184 

185 if lim == 0: 

186 terms.remove(term_two) 

187 return big_o(terms) 

188 

189 return big_o(terms[1:])

192class Timeout: 

193 """ 

194 Set a maximum amount of a time a block of code can take to execute. 

195 

196 Allows us to run a block of code such that a TimeoutError is thrown if 

197 it does not finish within a given amount of time. 

198 """ 

199 

200 def __init__(self, seconds=1, error_message='Timeout') -> None: 

201 """Create a new instance of Timeout.""" 

202 self.seconds = seconds 

203 self.error_message = error_message 

204 

205 def handle_timeout(self, signum, frame): 

206 """Execute after self.seconds have passed if the block within it is not done.""" 

207 raise TimeoutError(self.error_message) 

208 

209 def __enter__(self) -> None: 

210 """Start the timer when we begin executing the code within the block this class wraps.""" 

211 signal.signal(signal.SIGALRM, self.handle_timeout) 

212 signal.alarm(self.seconds) 

213 

214 def __exit__(self, err_type, value, traceback) -> None: 

215 """Stop the timer once the code block is done executing.""" 

216 signal.alarm(0)
