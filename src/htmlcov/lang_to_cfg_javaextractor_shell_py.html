15 """Functions like a basic bash shell."""

18 """Create a new instance of Shell.""" 

19 self.log = Log(tag=TAG)

27 def runcmd(self, cmd, cwd=None, shell=False): 

28 """Run a command in the shell.""" 

29 self.log.v_msg("cmd: {}\n with params: cwd={}, shell={}".format(cmd, cwd, shell)) 

30 args = shlex.split(cmd) 

31 process = Popen(args, stdout=PIPE, stderr=PIPE, cwd=cwd, shell=shell) 

32 out, err = process.communicate() 

33 if out: 

34 self.log.v_msg("cmd output: {}\n".format(out)) 

35 if err: 

36 self.log.v_msg("cmd output: {}\n".format(err)) 

37 

38 return out, err 

39 

40 def runcmd_bgrnd(self, cmd, out=PIPE, cwd=None, shell=False): 

41 """Run a command in a new process to prevent blocking.""" 

42 cmd_to_log = "cmd: {}\n with params: out={}, cwd={}, shell={}".format( 

43 cmd, out, cwd, shell) 

44 self.log.v_msg(cmd_to_log) 

45 redirect_to = out 

46 if out is not PIPE: 

47 redirect_to = open(out, "w") 

48 args = shlex.split(cmd) 

49 process = Popen(args, stdout=redirect_to, stderr=redirect_to, 

50 cwd=cwd, shell=shell) 

51 self.current_process = process 

52 self.process_output = redirect_to 

53 return process 

54 

55 def kill(self, process=None): 

56 """Kill the specificed process (or current process if not specified).""" 

57 if process is None: 

58 process = self.current_process 

59 

60 # Assign to _ to avoid linter issues. 

61 _ = process and process.kill() 

62 _ = self.process_output and self.process_output.close() 

63 self.current_process = None 

64 

65 def terminate(self, process=None): 

66 """Alias for self.kill.""" 

67 self.kill(process)
