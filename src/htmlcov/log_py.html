11class Colors(Enum):

24class LogLevel(Enum):

27 DEBUG = auto()

31class Log: 

32 """Log is used to print messages to standard out with additional formatting.""" 

33 

34 def __init__(self, tag: str = "", display_output: bool = True, 

35 log_level=LogLevel.REGULAR) -> None: 

36 """Create a new logger.""" 

37 self.tag = tag 

38 self.display_output = display_output 

39 self.log_level = log_level 

40 

41 def set_tag(self, tag: str) -> None: 

42 """Set the tag for the logger. This tag will be prepended to all messages.""" 

43 self.tag = tag 

44 

45 def d_msg(self, *msgs: str) -> None: 

46 """Use for debugging. Messages are only displayed if LOG_LEVEL is 'DEBUG'.""" 

47 if self.display_output and self.log_level is LogLevel.DEBUG: 

48 for msg in msgs: 

49 print(f"{Colors.OKBLUE.value} === DEBUG === {Colors.ENDC.value}: {msg}") 

50 

51 def i_msg(self, *msgs: str) -> None: 

52 """Log an information-level message.""" 

53 if self.display_output: 

54 for msg in msgs: 

55 print(f"{self.tag} ~~~ {msg.upper()} ~~~") 

56 

57 def v_msg(self, *msgs: str) -> None: 

58 """Log a verbose-level message.""" 

59 if self.display_output: 

60 for msg in msgs: 

61 print(f"{self.tag} > {msg}") 

62 

63 def e_msg(self, *msgs: str) -> None: 

64 """Log an error-level message.""" 

65 if self.display_output:
