1""" 

2Convert AST to CFG recursively. 

3 

4The idea is to iterate line by line. Lines can be either control-flow relevant or not. 

5If they are not relevant to control flow, we simply ignore them. 

6Otherwise, we will call the function recursively. Essentially, when we see a new control-flow 

7breaking node, we connect it to the previous node, then call it recursively, and it should make 

8all of the connections. It will return ALL of its leaf nodes that should be connected to the next 

9thing after the statement returns. 

10"""

29class FunctionVisitor(ast.NodeVisitor): 

30 """ 

31 FunctionVisitor handles the traversal through the AST for a single function. 

32 

33 It includes visitor functions for each type of statement we are interested in. 

34 """ 

35 

36 def __init__(self, logger=None) -> None: 

37 """Create a new instance of the function visitor.""" 

38 self.root: Optional[Node] = None 

39 self.end_node: Optional[Node] = None 

40 self.frontier: List[Node] = [] 

41 self.logger = logger 

42 

43 def update_root(self, node) -> bool: 

44 """Given a new node, set it as the root if a root does not exist.""" 

45 if self.root is None: 

46 self.root = node 

47 self.end_node = node 

48 return True 

49 

50 return False 

51 

52 def update_frontier(self, node) -> None: 

53 """Given a next node, set it as the child of all nodes in the frontier.""" 

54 for frontier_node in self.frontier: 

55 frontier_node.children += [node] 

56 if frontier_node == self.end_node: 

57 self.end_node = node

141 def visit_If(self, node) -> None: 

142 """Visit a python if statement.""" 

143 print(f"At if {node}") 

144 new_node = Node() 

145 if not self.update_root(new_node): 

146 self.update_frontier(new_node) 

147 

148 self.frontier = [new_node] 

149 

150 # Keep track of elements that will become the new frontier once we have looked 

151 # at ALL the if statements 

152 new_frontier = [] 

153 

154 # This is the comparison in the if statement 

155 print(f"{dir(node.test)}") 

156 

157 # Visit the body of the first if block 

158 visitor = FunctionVisitor() 

159 visitor.visit(node.body[0]) 

160 for frontier_node in self.frontier: 

161 if visitor.root is not None: 

162 frontier_node.children += [visitor.root] 

163 

164 if visitor.end_node is not None: 

165 new_frontier += [visitor.end_node] 

166 

167 i = 0 

168 # Check that there is another elif (or else statement) 

169 # Then, check that it is actually of type 'If', as the orself for an 

170 # else statement will actually get the body of the else 

171 # (see visit_If @ https://github.com/python/cpython/blob/master/Lib/ast.py) 

172 while len(node.orelse) != 0 and isinstance(node.orelse[0], ast.If): 

173 node = node.orelse[0] 

174 print(f"At the {i}th elif: {node}") 

175 # Visit the body of this elif block 

176 visitor = FunctionVisitor() 

177 visitor.visit(node.body[0]) 

178 for frontier_node in self.frontier: 

179 if visitor.root is not None: 

180 frontier_node.children += [visitor.root] 

181 if visitor.end_node is not None: 

182 new_frontier += [visitor.end_node] 

183 i += 1 

184 

185 # Check if there is an else statement 

186 if len(node.orelse) != 0: 

187 # Get the body of the else 

188 node = node.orelse 

189 visitor = FunctionVisitor() 

190 visitor.visit(node[0]) 

191 for frontier_node in self.frontier: 

192 if visitor.root is not None: 

193 frontier_node.children += [visitor.root] 

194 if visitor.end_node is not None: 

195 new_frontier += [visitor.end_node] 

196 

197 self.frontier = new_frontier

236class Visitor(ast.NodeVisitor): 

237 """ 

238 Visitor is used to look at Python source code. 

239 

240 It creates a FunctionVisitor for each function in the class. 

241 """ 

242 

243 def __init__(self) -> None: 

244 """Create a new instance of the Python source code parser.""" 

245 self.graphs: Dict[str, Graph] = {} 

246 

247 def visit_FunctionDef(self, node) -> None: 

248 """Call this for each function in the python source file.""" 

249 visitor = FunctionVisitor() 

250 print(f"Visiting {node.name}") 

251 visitor.visit(node) 

252 

253 # Now take the representation and convert it to a graph.py by doing BFS. 

254 

255 nodes_to_visit = [visitor.root] 

256 nodes = dict() # Maps node object to node number. 

257 nodes[visitor.root] = 0 

258 

259 edge_list = [] 

260 node_list = [0] 

261 

262 visited = set() 

263 

264 while len(nodes_to_visit) != 0: 

265 curr_node = nodes_to_visit[0] 

266 nodes_to_visit = nodes_to_visit[1:] 

267 if curr_node in visited: 

268 continue 

269 

270 visited.add(curr_node) 

271 curr_node = cast(Node, curr_node) 

272 children = curr_node.children 

273 

274 # Create all of the edges we need to from the current node to its children. 

275 for child in children: 

276 # Check if we need to create a new node. 

277 if child not in nodes: 

278 nodes[child] = len(node_list) 

279 node_list.append(len(node_list)) 

280 

281 edge_list.append((nodes[curr_node], nodes[child])) 

282 

283 nodes_to_visit += children 

284 

285 if len(visitor.frontier) != 1: 

286 print("Frontier has more than one node.") 

287 new_node = Node() 

288 nodes[new_node] = len(node_list) 

289 node_list.append(len(node_list)) 

290 visitor.end_node = new_node 

291 for frontier_node in visitor.frontier: 

292 edge_list.append((nodes[frontier_node], nodes[new_node])) 

293 

294 graph = Graph(edge_list, node_list, nodes[visitor.root], 

295 nodes[visitor.end_node], GraphType.EDGE_LIST) 

296 

297 self.graphs[node.name] = graph

308 def to_graph(self, filename: str, file_extension: str) -> Dict[str, Graph]: 

309 """Create a CFG from a Python source file.""" 

310 self.logger.d_msg(file_extension) 

311 

312 path = os.path.join(os.getcwd(), filename) + ".py" 

313 self.logger.d_msg(path) 

314 visitor = Visitor() 

315 graphs: Dict[str, Graph] 

316 with open(path, "r") as src: 

317 root = ast.parse(src.read()) 

318 visitor.visit(root) 

319 graphs = visitor.graphs 

320 

321 for key in graphs.keys(): 

322 self.logger.d_msg(f"==== Graph {key} =====") 

323 self.logger.d_msg(graphs[key]) 

324 

325 return graphs
